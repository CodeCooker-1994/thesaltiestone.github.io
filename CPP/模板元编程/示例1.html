    # 根据类型集合对类模板偏特化

    • DEMO 1
    #include <type_traits>

template <typename T>
struct IsVoidOrInt : std::false_type {};

template <>
struct IsVoidOrInt<int> : std::true_type {};

template <>
struct IsVoidOrInt<void> : std::true_type {};

template<typename T, typename = typename IsVoidOrInt<T>::type>
class A {
};

template<typename T>
class A<T, std::true_type> {
public:
    A() {std::cout << "T is void or int" << std::endl;}
};

template<typename T>
class A<T, std::false_type> {
public:
    A() {std::cout << "T is not void or int" << std::endl;}
};
    • DEMO 2
    #include <type_traits>

template<typename T>
struct IsVoidOrInt {
    static constexpr bool value = std::is_void<T>::value || std::is_same<T, int>::value;
};

template<typename T, bool = IsVoidOrInt<T>::value>
class A {
};

template<typename T>
class A<T, true> {
public:
    A() {std::cout << "T is void or int" << std::endl;}
};

template<typename T>
class A<T, false> {
public:
    A() {std::cout << "T is not void or int" << std::endl;}
};
