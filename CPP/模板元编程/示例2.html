    # 模板函数“偏特化”

    C++不支持模板函数偏特化，所以只能使用重载来进行模拟，但是如果直接用重载，就需要对具体的参数进行实例化，但是这在某些场景下并不容易做到，下面是通用方法。

    #include <vector>
#include <iostream>
#include <array>

enum class OverwriteType : uint8_t {
    COMMON = 0,
    VERTOR = 1,
    ARRAY = 2
};

template<OverwriteType v>
using OverwriteTypeConstant = std::integral_constant<OverwriteType, v>;

using VectorType = OverwriteTypeConstant<OverwriteType::VERTOR>;
using ArrayType = OverwriteTypeConstant<OverwriteType::ARRAY>;
using CommonType = OverwriteTypeConstant<OverwriteType::COMMON>;

template<typename T>
struct WhichOverwriteType : public CommonType {
};

template<typename T>
struct WhichOverwriteType<std::vector<T>> : public VectorType {
};

template<typename T, size_t N>
struct WhichOverwriteType<std::array<T, N>> : public ArrayType {
};

template<typename U>
void func(VectorType, U u)
{
    std::cout << "vector, u = " << u << std::endl;
}

template<typename U>
void func(ArrayType, U u)
{
    std::cout << "array, u = " << u << std::endl;
}

template<typename U>
void func(CommonType, U u)
{
    std::cout << "common, u = " << u << std::endl;
}

template<typename T, typename U>
void func(U u)
{
    func<U>(WhichOverwriteType<T>{}, u);
}

int main()
{
    func<int>(1);
    func<std::vector<int>>(1);
    func<std::array<int, 5>>(1);
    return 0;
}
